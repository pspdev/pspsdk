37#define pspKernelSetKernelPC() \
38{ \
39 __asm__ volatile ( \
40 "la $8, 1f\n\t" \
41 "lui $9, 0x8000\n\t" \
42 "or $8, $9\n\t" \
43 "jr $8\n\t" \
44 " nop\n\t" \
45 "1:\n\t" \
46 : : : "$8", "$9"); \
47 sceKernelIcacheClearAll(); \
48}
