96/*
97 * Local port number conventions:
98 *
99 * Ports < IPPORT_RESERVED are reserved for privileged processes (e.g. root),
100 * unless a kernel is compiled with IPNOPRIVPORTS defined.
101 *
102 * When a user does a bind(2) or connect(2) with a port number of zero,
103 * a non-conflicting local port address is chosen.
104 *
105 * The default range is IPPORT_ANONMIN to IPPORT_ANONMAX, although
106 * that is settable by sysctl(3); net.inet.ip.anonportmin and
107 * net.inet.ip.anonportmax respectively.
108 *
109 * A user may set the IPPROTO_IP option IP_PORTRANGE to change this
110 * default assignment range.
111 *
112 * The value IP_PORTRANGE_DEFAULT causes the default behavior.
113 *
114 * The value IP_PORTRANGE_HIGH is the same as IP_PORTRANGE_DEFAULT,
115 * and exists only for FreeBSD compatibility purposes.
116 *
117 * The value IP_PORTRANGE_LOW changes the range to the "low" are
118 * that is (by convention) restricted to privileged processes.
119 * This convention is based on "vouchsafe" principles only.
120 * It is only secure if you trust the remote host to restrict these ports.
121 * The range is IPPORT_RESERVEDMIN to IPPORT_RESERVEDMAX.
122 */
123
124#define IPPORT_RESERVED 1024
125#define IPPORT_ANONMIN 49152
126#define IPPORT_ANONMAX 65535
127#define IPPORT_RESERVEDMIN 600
128#define IPPORT_RESERVEDMAX (IPPORT_RESERVED-1)
